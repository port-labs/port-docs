          title: .folder.name

          properties:
            url: >-
              .repo.web_url + "/tree/" + .repo.default_branch  + "/" +
              .folder.path
            description: .repo.description
            namespace: .repo.namespace.name
            full_path: .repo.path_with_namespace + "/" + folder.path
            language: .repo.__languages | to_entries | max_by(.value) | .key
